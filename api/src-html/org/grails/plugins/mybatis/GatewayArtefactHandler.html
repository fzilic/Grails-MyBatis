001package org.grails.plugins.mybatis;
002
003import org.apache.commons.logging.Log;
004import org.apache.commons.logging.LogFactory;
005import org.codehaus.groovy.grails.commons.ArtefactHandlerAdapter;
006import org.codehaus.groovy.grails.commons.GrailsClass;
007
008public class GatewayArtefactHandler extends ArtefactHandlerAdapter {
009  public static final String TYPE = "Gateway";
010  private static Log log = LogFactory.getLog(GatewayArtefactHandler.class);
011
012  public GatewayArtefactHandler() {
013    super(TYPE, GrailsClass.class, DefaultGrailsGatewayClass.class, null);
014    log.debug("Created instance");
015  }
016
017  @Override
018  @SuppressWarnings("rawtypes")
019  public boolean isArtefactClass(Class clazz) {
020    log.debug("Checking potential artefact class " + clazz);
021    return clazz != null && clazz.getName().endsWith(TYPE);
022  }
023
024  @Override
025  public boolean isArtefactGrailsClass(GrailsClass grailsClass) {
026    log.debug("Checking potential artefact class " + grailsClass + " within isArtefactGrailsClass");
027    return super.isArtefactGrailsClass(grailsClass);
028  }
029}
